# 4-2 「A」の扱いについて

ブラックジャックのルールとして「A」は「1」と「11」を選ぶというルールがあります。このパートではそれをどう表すか、という点にスポットを当てていきます。

では一度、下記のように修正してください。  
解説はコード内に記載しておきます。

`blackjack.rb`

~~~ruby
require "./deck"
require "./card"
require "./player"
require "./dealer"

class Blackjack

  def start
    build_player
    build_deck
    build_dealer

    while true
      #変数を定義
      @count_11 = 0
      @count_11_dealer = 0

      @dealer.first_draw_dealer(@deck)
      @dealer_point = point_dealer
      @player.first_draw_player(@deck)
      @player_point = point_player
      #@count_11が0か1によって表示するメッセージを使い分けます。
      if @count_11 == 0
        puts <<~text
        あなたの手札の合計点数は#{@player_point}です。
        ----------------------------------
        text

      else
        puts <<~text
        あなたの手札の合計点数は#{@player_point}、もしくは#{@player_point-10}です。
        ----------------------------------
        text
      end

      while true
        puts <<~text
              あなたの行動を選択してください
              1.Hit 2.Stand
              text
        # ユーザの入力数値をaction変数に代入
        action = gets.chomp.to_i
        if action == 1
          # ユーザがHitを選択。デッキからカードを引く
          @player.draw_player(@deck)
          # playerのポイントを計算
          @player_point = point_player
          #再度手札を引いた際も@count_11の数によって「A」の扱いを決める
          if @count_11 == 0
            puts <<~text
                あなたの手札の合計点数は#{@player_point}です。
                ----------------------------------
                text
          else
            puts <<~text
                あなたの手札の合計点数は#{@player_point}、もしくは#{@player_point-10}です。
                ----------------------------------
                text
          end

        elsif action == 2
          break

        else
          puts <<~text
                ----------------------------------
                error ： 1か2を入力してください
                ----------------------------------
                text
        end
      end
      break
    end
  end

  private
    def build_player
      @player = Player.new
    end

    def build_deck
      @deck = Deck.new
    end

    def build_dealer
      @dealer = Dealer.new
    end

    def point_player
      #点数の初期化
      player_point = 0
      count_a = 0

      @player.hands.each do |hand|
        player_point += point(hand)
        #「A」が何回出たかを計算
        if point(hand) == 0
          count_a += 1
        end
      end
      #player_pointが10以下の時、10以上の時で場合わけし、timesメソッドにより、変数@count_11の数値を決める。
      count_a.times do |i|
        if player_point <= 10
          player_point += 11
          @count_11 = 1
        else
          player_point += 1
          @count_11 = 0
        end
      end
      #puts "playerの得点は#{@player_point}点です" このパートで消します。
      return player_point
    end
    #`point_playerと意味は同じです`
    def point_dealer
      dealer_point = 0
      count_a = 0

      @dealer.hands.each do |hand|
        dealer_point += point(hand)
        if point(hand) == 0
          count_a += 1
        end
      end

      count_a.times do |i|
        if dealer_point <= 10
          dealer_point += 11
          @count_11_dealer = 1
        else
          dealer_point += 1
          @count_11_dealer = 0
        end
      end

      puts "dealererの得点は#{dealer_point}点です"
      dealer_point
    end

    def point(card)
      if card.number == "J" || card.number == "Q" || card.number == "K"
        return number = 10
      elsif card.number == "A"
        #「1」→「0」へ修正
        return number = 0
      else
        return card.number.to_i
      end
    end

end
~~~
